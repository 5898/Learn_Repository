### 快速排序
时间复杂度O( nlogn )
最坏的情况 时间复杂度O( n^2 )
```javascript
// 快速排序
function quickOrder(arr) {
  if (arr.length <= 1) return arr;

  let left = [], right = [];
  let temp = arr.splice(0, 1)[0];
  for (let i = 0; i < arr.length; i++){
    if (arr[i] <=temp) {
      left.push(arr[i])
    } else{
      right.push(arr[i])
    }
  }
  return quickOrder(left).concat(temp,quickOrder(right));
}
```
